purge_aiseirigh = {
	if = {
		limit = {
			has_idea = FHI_aiseirigh_threat
		}
		add_stability = -0.2
	}
	if = {
		limit = {
			has_idea = FHI_imminent_aiseirigh_threat
		}
		add_stability = -0.4
	}
	if = {
		limit = {
			has_idea = FHI_working_with_aiseirigh
		}
		add_stability = -0.2
	}
	if = {
		limit = {
			has_idea = FHI_aiseirigh_security_forces
		}
		add_stability = -0.4
	}
	if = {
		limit = {
			has_idea = FHI_aiseirigh_out_of_control
		}
		add_stability = -0.4
	}
	set_country_flag = aiseirigh_over
	set_political_party = {
		ideology = supremacism
		popularity = 0
	}
	remove_aiseirigh_ideas = yes
}

close_down_aiseirigh = {
	set_country_flag = aiseirigh_over
	set_political_party = {
		ideology = supremacism
		popularity = 0
	}
	remove_aiseirigh_ideas = yes
}

FHI_improve_govt_unity = {
	if = {
		limit = {
			has_idea = FHI_contested_commission
		}
		swap_ideas = {
			remove_idea = FHI_contested_commission
			add_idea = FHI_contested_commission_2
		}
	}
	if = {
		limit = {
			has_idea = FHI_contested_commission_2
		}
		remove_ideas = FHI_contested_commission_2
	}
}

FHI_decrease_govt_unity = {
	if = {
		limit = {
			has_idea = FHI_contested_commission_2
		}
		swap_ideas = {
			remove_idea = FHI_contested_commission_2
			add_idea = FHI_contested_commission
		}
	}
	else = {
		add_ideas = FHI_contested_commission_2
	}
}
