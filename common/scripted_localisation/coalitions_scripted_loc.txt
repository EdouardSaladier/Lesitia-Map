defined_text = {
	name = GetCoalitionMemberBG_despotism
	text = {
		trigger = {
			NOT = {
				has_government = despotism
			}
			has_country_flag = coalition_despotism
		}
		localization_key = "GFX_coalition_member"
	}
	text = {
		trigger = {
			OR = {
				NOT = {
					has_country_flag = coalition_despotism
				}
				has_government = despotism
			}
		}
		localization_key = "GFX_coalition_no_bg"
	}
}
