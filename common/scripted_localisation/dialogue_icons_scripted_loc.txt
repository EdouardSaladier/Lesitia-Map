defined_text = {
	name = GetChar1Icon
	text = { # Char 1 talking
		trigger = {
			has_country_flag = char1_talking
		}
		localization_key = "GFX_dialogue_char1"
	}
	text = { # Char 1 NOT talking
		trigger = {
			NOT = { has_country_flag = char1_talking }
		}
		localization_key = "GFX_dialogue_char1d"
	}
}

	name = GetCharName

		trigger = {
			has_country_flag = char1_talking
		}
		localization_key = "char_name1"
	}
	text = { # Char 2 talking
		trigger = {
			NOT = { has_country_flag = char1_talking }
			has_country_flag = char2_talking
		}
		localization_key = "char_name2"
	}
	text = { # Nobody talking
		trigger = {

		}

	}
