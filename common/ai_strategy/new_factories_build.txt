build_electronics_factories = {
	enable = {
		OR = {
			check_variable = { resource@electronics < 0 }
			check_variable = { resource_imported@electronics > 16.0 }
		}
	}
	
	abort_when_not_enabled = yes

	ai_strategy = {
		type = building_target
		id = electronics_factory
		value = 100
	}
}
